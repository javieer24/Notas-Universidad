
Convertir el algorítmo visto en clase en un algorítmo recursivo
En pseudocodigo orientado a java
public int buscar(int[] array, int valor) {
  for (int i = 0; i < array.length; i++) {
    if (array[i] == valor) {
      return i;
    }
  }
  return -1; // Si no se encuentra el valor, se devuelve -1
}

public int buscarRecursivo(int[] array, int valor, int indice) {
  if (indice == array.length) {
    return -1; // Si no se encuentra el valor, se devuelve -1
  } else if (array[indice] == valor) {
    return indice;
  } else {
    return buscarRecursivo(array, valor, indice + 1);
  }
}

-En java código 
public class BusquedaRecursiva {

    public static int buscar(int[] array, int valor) {
        return buscarRecursivo(array, valor, 0);
    }

    private static int buscarRecursivo(int[] array, int valor, int indice) {
        if (indice == array.length) {
            return -1; // Si no se encuentra el valor, se devuelve -1
        } else if (array[indice] == valor) {
            return indice;
        } else {
            return buscarRecursivo(array, valor, indice + 1);
        }
    }
    
    public static void main(String[] args) {
        int[] arreglo = {10, 20, 30, 40, 50};
        int valorBuscado = 30;
        int indiceEncontrado = buscar(arreglo, valorBuscado);
        if (indiceEncontrado == -1) {
            System.out.println("El valor " + valorBuscado + " no se encontró en el arreglo.");
        } else {
            System.out.println("El valor " + valorBuscado + " se encontró en el índice " + indiceEncontrado + " del arreglo.");
        }
    }
}
