
clase Pedido {
  atributos:
    - cliente: Cliente
    - fecha: Fecha
    - productos: Lista<Producto>
    - estado: EstadoPedido
    - formaDePago: FormaDePago

  métodos:
    + calcularCosteTotal(): Dinero
    + informarExistencias(): void
}

clase Producto {
  atributos:
    - nombre: Cadena
    - precio: Dinero
    - cantidad: Entero
    - impuestos: Dinero

  métodos:
    + obtenerNombre(): Cadena
}

clase FormaDePago {
  atributos:
    - tipo: TipoFormaDePago
    - fechaCaducidad: Fecha
    - numero: Cadena
    - moneda: Moneda
    - nombreCheque: Cadena
    - entidadBancariaCheque: Cadena

  métodos:
    + obtenerTipo(): TipoFormaDePago
}

clase Cliente {
  atributos:
    - nombre: Cadena
    - direccion: Cadena

  métodos:
    + obtenerNombre(): Cadena
}

clase EstadoPedido {
  atributos:
    - pendiente: Booleano
    - pagado: Booleano
    - procesando: Booleano
    - enviado: Booleano
    - entregado: Booleano

  métodos:
    + obtenerEstado(): EstadoPedido
